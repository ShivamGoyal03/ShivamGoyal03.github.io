class DocumentationValidator:
  def __init__(self, mcp_server, copilot_integration):
    self.mcp_server = mcp_server
    self.copilot_integration = copilot_integration

  async def validate_document_references(self, document):
    references = self.extract_references(document)
    return await self.mcp_server.validate_against_mslearn(references)

  async def generate_citations(self, content: str):
    return await self.mcp_server.generate_mslearn_citations(content)
